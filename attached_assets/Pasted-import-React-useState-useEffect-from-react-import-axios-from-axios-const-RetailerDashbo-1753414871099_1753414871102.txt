import React, { useState, useEffect } from 'react';
import axios from 'axios';

const RetailerDashboard = () => {
  const [sales, setSales] = useState(0);
  const [fee, setFee] = useState(0);
  const [walletBalance, setWalletBalance] = useState(0);
  const [adBudget, setAdBudget] = useState(100);

  const userId = 'demo-user-001';

  useEffect(() => {
    fetchWallet();
  }, []);

  const fetchWallet = async () => {
    try {
      const res = await axios.get(`/api/spiral-wallet/${userId}`);
      setWalletBalance(res.data.balance);
    } catch (err) {
      console.error('Wallet fetch error', err);
    }
  };

  const handleSalesChange = (e: React.ChangeEvent<HTMLInputElement>) => {
    const value = parseFloat(e.target.value);
    setSales(value);
    setFee(calculateFee(value));
  };

  const calculateFee = (amount: number) => {
    if (amount <= 1000) return amount * 0.05;
    if (amount <= 5000) return amount * 0.06;
    return amount * 0.07;
  };

  const handleAdBudgetChange = (e: React.ChangeEvent<HTMLInputElement>) => {
    setAdBudget(parseFloat(e.target.value));
  };

  return (
    <div className="p-6 bg-white shadow-md rounded-xl max-w-xl mx-auto mt-10">
      <h1 className="text-2xl font-bold mb-4 text-center">Retailer Dashboard</h1>

      <div className="mb-4">
        <label className="block font-medium mb-1">Monthly Sales ($)</label>
        <input
          type="number"
          value={sales}
          onChange={handleSalesChange}
          className="w-full p-2 border rounded"
        />
        <p className="mt-2 text-gray-700">SPIRAL Platform Fee: <strong>${fee.toFixed(2)}</strong></p>
      </div>

      <div className="mb-4">
        <label className="block font-medium mb-1">SPIRAL Wallet Balance</label>
        <div className="p-2 border rounded bg-green-50 font-semibold">${walletBalance.toFixed(2)}</div>
      </div>

      <div className="mb-4">
        <label className="block font-medium mb-1">Ad Budget (Preview Tool)</label>
        <input
          type="number"
          value={adBudget}
          onChange={handleAdBudgetChange}
          className="w-full p-2 border rounded"
        />
        <p className="mt-2 text-gray-700">
          Estimated Reach: <strong>{(adBudget * 50).toLocaleString()} impressions</strong>
        </p>
      </div>

      <div className="text-center">
        <button
          onClick={fetchWallet}
          className="px-6 py-2 bg-blue-600 text-white rounded hover:bg-blue-700"
        >
          Refresh Wallet
        </button>
      </div>
    </div>
  );
};

export default RetailerDashboard;
